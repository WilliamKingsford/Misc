01063 struct CollectParam {
01064     int offset;
01065     int limit;
01066     int count;
01067     GList *commits;
01068 #ifdef SEAFILE_SERVER
01069     gint64 truncate_time;
01070     gboolean traversed_head;
01071 #endif
01072 };
01073 
01074 static gboolean
01075 get_commit (SeafCommit *c, void *data, gboolean *stop)
01076 {
01077     struct CollectParam *cp = data;
01078 
01079 #ifdef SEAFILE_SERVER
01080     if (cp->truncate_time == 0)
01081     {
01082         *stop = TRUE;
01083         /* Stop after traversing the head commit. */
01084     }
01085     /* We use <= here. This is for handling clean trash and history.
01086      * If the user cleans all history, truncate time will be equal to
01087      * the commit's ctime. In such case, we don't actually want to display
01088      * this commit.
01089      */
01090     else if (cp->truncate_time > 0 &&
01091              (gint64)(c->ctime) <= cp->truncate_time &&
01092              cp->traversed_head)
01093     {
01094         *stop = TRUE;
01095         return TRUE;
01096     }
01097 
01098     /* Always traverse the head commit. */
01099     if (!cp->traversed_head)
01100         cp->traversed_head = TRUE;
01101 #endif
01102 
01103     /* if offset = 1, limit = 1, we should stop when the count = 2 */
01104     if (cp->limit > 0 && cp->count >= cp->offset + cp->limit) {
01105         *stop = TRUE;
01106         return TRUE;  /* TRUE to indicate no error */
01107     }
01108 
01109     if (cp->count >= cp->offset) {
01110         SeafileCommit *commit = convert_to_seafile_commit (c);
01111         cp->commits = g_list_prepend (cp->commits, commit);
01112     }
01113 
01114     ++cp->count;
01115     return TRUE;                /* TRUE to indicate no error */
01116 }

03119                     GError **error)
03120 {
03121     SeafRepo *repo = NULL;
03122     SeafCommit *commit = NULL;
03123     char *obj_id = NULL;
03124 
03125     if (!repo_id || !path) {
03126         g_set_error (error, SEAFILE_DOMAIN, SEAF_ERR_BAD_ARGS,
03127                      "Bad arguments");
03128         return NULL;

03131     if (!is_uuid_valid (repo_id)) {

03133         return NULL;

03137     if (!repo) {
03138         g_set_error (error, SEAFILE_DOMAIN, SEAF_ERR_INTERNAL,
03139                      "Get repo error");
03140         goto out;
03141     }
03142 
03143     commit = seaf_commit_manager_get_commit (seaf->commit_mgr,
03144                                              repo->id, repo->version,
03145                                              repo->head->commit_id);

03147         g_set_error (error, SEAFILE_DOMAIN, SEAF_ERR_INTERNAL,

03154                                              repo->store_id, repo->version,
03155                                              commit->root_id,
03156                                              path, &mode, error);
03157 
03158 out:
03159     if (repo)
03160         seaf_repo_unref (repo);
03161     if (commit)
03162         seaf_commit_unref (commit);
03163     if (obj_id) {
03164         /* check if the mode matches */
03165         if ((want_dir && !S_ISDIR(mode)) || ((!want_dir) && S_ISDIR(mode))) {
03166             g_free (obj_id);
03167             return NULL;
03168         }
03169     }
03170

03313 /*

03316 static gint
03317 comp_dirent_func (gconstpointer a, gconstpointer b)
03318 {
03319     const SeafDirent *dent_a = a, *dent_b = b;
03320 
03321     if (S_ISDIR(dent_a->mode) && S_ISREG(dent_b->mode))
03322         return -1;
03323 
03324     if (S_ISREG(dent_a->mode) && S_ISDIR(dent_b->mode))
03325         return 1;
