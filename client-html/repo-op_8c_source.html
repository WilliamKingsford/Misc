02395 static char *
02396 rename_file_recursive(SeafRepo *repo,
02397                       const char *dir_id,
02398                       const char *to_path,
02399                       const char *oldname,
02400                       const char *newname)
02401 {
02402     SeafDir *olddir, *newdir;
02403     SeafDirent *dent;
02404     GList *ptr;
02405     char *to_path_dup = NULL;
02406     char *remain = NULL;
02407     char *slash;
02408     char *id = NULL;
02409 
02410     olddir = seaf_fs_manager_get_seafdir_sorted(seaf->fs_mgr,
02411                                                 repo->store_id, repo->version,
02412                                                 dir_id);
02413     if (!olddir)
02414         return NULL;
02415 
02416     /* we reach the target dir. */
02417     if (*to_path == '\0') {
02418         SeafDirent *old, *newdent = NULL;
02419         GList *newentries = NULL, *p;
02420 
02421         /* When renameing, there is a pitfall: we can't simply rename the
02422          * dirent, since the dirents are required to be sorted in descending
02423          * order. We need to copy all old dirents except the target dirent,
02424          * and then rename the target dirent, and then insert the new
02425          * dirent, so that we can maintain the descending order of dirents. */
02426         for (p = olddir->entries; p != NULL; p = p->next) {
02427             old = p->data;
02428             if (strcmp(old->name, oldname) != 0) {
02429                 newentries = g_list_prepend (newentries, seaf_dirent_dup(old));
02430             } else {
02431                 newdent = seaf_dirent_new (old->version, old->id, old->mode,
02432                                            newname, old->mtime,
02433                                            old->modifier, old->size);
02434             }
02435         }
02436 
02437         newentries = g_list_reverse (newentries);
02438 
02439         if (newdent) {
02440             newentries = g_list_insert_sorted(newentries, newdent, compare_dirents);
02441         }
02442 
02443         newdir = seaf_dir_new (NULL, newentries,
02444                                dir_version_from_repo_version(repo->version));
02445         seaf_dir_save (seaf->fs_mgr, repo->store_id, repo->version, newdir);
02446         id = g_strndup (newdir->dir_id, 40);
02447         seaf_dir_free (newdir);
02448 
02449         goto out;
02450     }
02451 
02452     to_path_dup = g_strdup (to_path);
02453     slash = strchr (to_path_dup, '/');
02454 
02455     if (!slash) {
02456         remain = to_path_dup + strlen(to_path_dup);
02457     } else {
02458         *slash = '\0';
02459         remain = slash + 1;
02460     }
02461 
02462     for (ptr = olddir->entries; ptr; ptr = ptr->next) {
02463         dent = (SeafDirent *)ptr->data;
02464 
02465         if (strcmp(dent->name, to_path_dup) != 0)
02466             continue;
02467 
02468         id = rename_file_recursive (repo, dent->id, remain, oldname, newname);
02469         if (id != NULL) {
02470             memcpy(dent->id, id, 40);
02471             dent->id[40] = '\0';
02472         }
02473         break;
02474     }
02475     
02476     if (id != NULL) {
02477         /* Create a new SeafDir. */
02478         GList *new_entries;
02479         
02480         new_entries = dup_seafdir_entries (olddir->entries);
02481         newdir = seaf_dir_new (NULL, new_entries,
02482                                dir_version_from_repo_version(repo->version));
02483         seaf_dir_save (seaf->fs_mgr, repo->store_id, repo->version, newdir);
02484         
02485         g_free(id);
02486         id = g_strndup(newdir->dir_id, 40);
02487         seaf_dir_free (newdir);
02488     }
02489 
02490 out:
02491     g_free (to_path_dup);
02492     seaf_dir_free(olddir);
02493     return id;
02494 }

02586 static char *
02587 put_file_recursive(SeafRepo *repo,
02588                    const char *dir_id,
02589                    const char *to_path,

02591 {

02593     SeafDirent *dent;
02594     GList *ptr;

02599 
02600     olddir = seaf_fs_manager_get_seafdir_sorted(seaf->fs_mgr,
02601                                                 repo->store_id, repo->version,
02602                                                 dir_id);
02603     if (!olddir)
02604         return NULL;
02605 
02606     /* we reach the target dir. Update the target dirent. */
02607     if (*to_path == '\0') {
02608         GList *newentries = NULL, *p;
02609         SeafDirent *dent;
02610 
02611         for (p = olddir->entries; p; p = p->next) {
02612             dent = p->data;
02613             if (strcmp(dent->name, newdent->name) == 0) {
02614                 newentries = g_list_prepend (newentries, seaf_dirent_dup(newdent));
02615             } else {
02616                 newentries = g_list_prepend (newentries, seaf_dirent_dup(dent));
02617             }
02618         }
02619 
02620         newentries = g_list_reverse (newentries);
02621         newdir = seaf_dir_new (NULL, newentries,
02622                                dir_version_from_repo_version(repo->version));
02623         seaf_dir_save (seaf->fs_mgr, repo->store_id, repo->version, newdir);

02625         seaf_dir_free (newdir);
02626 
02627         goto out;
02628     }
02629 
02630     to_path_dup = g_strdup (to_path);
02631     slash = strchr (to_path_dup, '/');
02632 
02633     if (!slash) {
02634         remain = to_path_dup + strlen(to_path_dup);
02635     } else {
02636         *slash = '\0';
02637         remain = slash + 1;
02638     }
02639 
02640     for (ptr = olddir->entries; ptr; ptr = ptr->next) {
02641         dent = (SeafDirent *)ptr->data;
02642 
02643         if (strcmp(dent->name, to_path_dup) != 0)
02644             continue;
02645 
02646         id = put_file_recursive (repo, dent->id, remain, newdent);
02647         if (id != NULL) {
02648             memcpy(dent->id, id, 40);
02649             dent->id[40] = '\0';
02650             if (repo->version > 0)
02651                 dent->mtime = (guint64)time(NULL);
02652         }
02653         break;
02654     }
02655     
02656     if (id != NULL) {
02657         /* Create a new SeafDir. */
02658         GList *new_entries;
02659         
02660         new_entries = dup_seafdir_entries (olddir->entries);
02661         newdir = seaf_dir_new (NULL, new_entries,
02662                                dir_version_from_repo_version(repo->version));
02663         seaf_dir_save (seaf->fs_mgr, repo->store_id, repo->version, newdir);
02664         
02665         g_free(id);
02666         id = g_strndup(newdir->dir_id, 40);
02667         seaf_dir_free (newdir);
02668     }
02669 
02670 out:
02671     g_free (to_path_dup);
02672     seaf_dir_free(olddir);
02673     return id;
02674 }

03236                                old_dent->mtime, old_dent->modifier, old_dent->size);

03383     }
03384     
03385     if (!root_id) {
03386         seaf_warning ("[revert file] Failed to revert file.\n");

03389         ret = -1;
03390         goto out;
03391     }
03392

03400 #endif
03401     snprintf(buf, SEAF_PATH_MAX, "Reverted file \"%s\" to status at %s", filename, time_str);

03404         ret = -1;
03405         goto out;
03406     }
03407

03409 
03410 out:

03416         seaf_commit_unref (old_commit);

03419     g_free (parent_dir);
03420     g_free (filename);
03421 
03422     g_free (canon_path);
03423     seaf_dirent_free (old_dent);

03712 static gboolean
03713 collect_file_revisions (SeafCommit *commit, void *vdata, gboolean *stop)
03714 {
03715     CollectRevisionParam *data = vdata;
03716     SeafRepo *repo = data->repo;
03717     const char *path = data->path;
03718     GError **error = data->error;
03719     GHashTable *file_id_cache = data->file_id_cache;
03720 
03721     SeafCommit *parent_commit = NULL;
03722     SeafCommit *parent_commit2 = NULL;
03723     char *file_id = NULL;
03724     char *parent_file_id = NULL;
03725     char *parent_file_id2 = NULL;
03726     gint64 file_size;
03727 
03728     gboolean ret = TRUE;
03729 
03730     /* At least find the latest revision. */
03731     if (data->got_latest && data->truncate_time == 0) {
03732         *stop = TRUE;
03733         return TRUE;
03734     }
03735 
03736     if (data->got_latest &&
03737         data->truncate_time > 0 &&
03738         (gint64)(commit->ctime) < data->truncate_time)
03739     {
03740         *stop = TRUE;
03741         return TRUE;
03742     }
03743 
03744     if (data->max_revision > 0 && data->n_commits > data->max_revision) {
03745         *stop = TRUE;
03746         return TRUE;
03747     }
03748 
03749     file_id = get_file_id_with_cache (data->repo, commit, path,
03750                                       file_id_cache, error);
03751     if (*error) {
03752         ret = FALSE;
03753         goto out;
03754     }
03755 
03756     if (!file_id) {
03757         /* Target file is not present in this commit. */
03758         goto out;
03759     }
03760 
03761     file_size = seaf_fs_manager_get_file_size (seaf->fs_mgr,
03762                                                repo->store_id,
03763                                                repo->version,
03764                                                file_id);
03765 
03766     if (!commit->parent_id) {
03767         /* Initial commit */
03768         add_revision_info (data, commit, file_id, file_size);
03769         goto out;
03770     }
03771 
03772     parent_commit = seaf_commit_manager_get_commit (seaf->commit_mgr,
03773                                                     repo->id, repo->version, 
03774                                                     commit->parent_id);
03775     if (!parent_commit) {
03776         g_set_error (error, SEAFILE_DOMAIN, SEAF_ERR_GENERAL,
03777                      "Faild to get commit %s", commit->parent_id);
03778         ret = FALSE;
03779         goto out;
03780     }
03781 
03782     parent_file_id = get_file_id_with_cache (data->repo, parent_commit,
03783                                              path, file_id_cache, error);
03784     if (*error) {
03785         ret = FALSE;
03786         goto out;
03787     }
03788 
03789     if (g_strcmp0 (parent_file_id, file_id) == 0) {
03790         /* This commit does not modify the target file */
03791         goto out;
03792     }
03793 
03794     /* In case of a merge, the second parent also need compare */
03795     if (commit->second_parent_id) {
03796         parent_commit2 = seaf_commit_manager_get_commit (seaf->commit_mgr,
03797                                                          repo->id, repo->version, 
03798                                                          commit->second_parent_id);

03863 static gboolean
03864 detect_rename_revision (SeafRepo *repo,
03865                         SeafCommit *commit,
03866                         const char *path,
03867                         char **parent_id,
03868                         char **old_path)
03869 {
03870     GList *diff_res = NULL;
03871     SeafCommit *p1 = NULL;
03872     int rc;
03873     gboolean is_renamed = FALSE;
03874 
03875     while (*path == '/' && *path != 0)
03876         ++path;
03877 
03878     if (!commit->second_parent_id) {
03879         p1 = seaf_commit_manager_get_commit (seaf->commit_mgr,
03880                                              repo->id, repo->version,
03881                                              commit->parent_id);
03882         if (!p1) {
03883             seaf_warning ("Failed to get commit %.8s.\n", commit->parent_id);
03884             return FALSE;
03885         }
03886         /* Don't fold diff results for directories. We need to know a file was
03887          * renamed when its parent folder was renamed.
03888          */
03889         rc = diff_commits (p1, commit, &diff_res, FALSE);
03890         seaf_commit_unref (p1);
03891         if (rc < 0) {
03892             seaf_warning ("Failed to diff.\n");
03893             return FALSE;
03894         }
03895     } else {
03896         rc = diff_merge (commit, &diff_res, FALSE);
03897         if (rc < 0) {
03898             seaf_warning ("Failed to diff merge.\n");
03899             return FALSE;
03900         }
03901     }
03902 
03903     GList *ptr;
03904     DiffEntry *de;
03905     for (ptr = diff_res; ptr; ptr = ptr->next) {
03906         de = ptr->data;
03907         if (de->status == DIFF_STATUS_RENAMED && strcmp (de->new_name, path) == 0) {
03908             *old_path = g_strdup(de->name);
03909             is_renamed = TRUE;
03910             break;
03911         }
03912     }
03913     for (ptr = diff_res; ptr; ptr = ptr->next)
03914         diff_entry_free ((DiffEntry *)ptr->data);
03915     g_list_free (diff_res);
03916 
03917     if (!is_renamed)
03918         return FALSE;
03919 
03920     /* Determine parent commit containing the old path. */
03921     if (!commit->second_parent_id)
03922         *parent_id = g_strdup(commit->parent_id);
03923     else {
03924         if (path_exists_in_commit (repo, commit->parent_id, *old_path))
03925             *parent_id = g_strdup(commit->parent_id);
03926         else if (path_exists_in_commit (repo, commit->second_parent_id, *old_path))
03927             *parent_id = g_strdup(commit->second_parent_id);
03928         else {
03929             g_free (*old_path);
03930             *old_path = NULL;
03931             return FALSE;
03932         }
03933     }
03934 
03935     return TRUE;
03936 }

04106 static gboolean
04107 check_non_existing_files (void *key, void *value, void *vdata)
04108 {

04110     gboolean remove = FALSE;
04111     
04112     char *file_name = key;

04114     if (!ctime) {
04115         /* Impossible */
04116         remove = TRUE;
04117     } else if (*ctime != data->current_commit->ctime) {
04118         /* This file does not exist in this commit. So it's last modified in

04127 static gboolean
04128 collect_files_last_modified (SeafCommit *commit, void *vdata, gboolean *stop)
04129 {
04130     CalcFilesLastModifiedParam *data = vdata;
04131     GError **error = data->error;
04132     SeafDirent *dent = NULL;
04133     char *file_id = NULL;
04134     SeafDir *dir = NULL;
04135     GList *ptr;
04136     gboolean ret = TRUE;
04137 
04138     data->current_commit = commit;
04139     dir = seaf_fs_manager_get_seafdir_by_path (seaf->fs_mgr,
04140                                                data->repo->store_id,
04141                                                data->repo->version,
04142                                                commit->root_id,
04143                                                data->parent_dir,
04144                                                error);
04145     if (*error) {
04146         if (!g_error_matches(*error, SEAFILE_DOMAIN, SEAF_ERR_PATH_NO_EXIST)) {
04147             *stop = TRUE;
04148             ret = FALSE;
04149             goto out;
04150         } else {
04151             g_clear_error (error);
04152         }
04153     }
04154 
04155     if (!dir) {
04156         /* The directory does not exist in this commit. So all files are last
04157          * modified in the previous commit;
04158          */
04159         *stop = TRUE;
04160         goto out;
04161     }
04162 
04163     for (ptr = dir->entries; ptr; ptr = ptr->next) {
04164         dent = ptr->data;
04165         file_id = g_hash_table_lookup (data->current_file_id_hash, dent->name);
04166         if (file_id) {
04167             if (strcmp(file_id, dent->id) != 0) {
04168                 g_hash_table_remove (data->current_file_id_hash, dent->name);
04169             } else {
04170                 gint64 *ctime = g_new (gint64, 1);
04171                 *ctime = commit->ctime;
04172                 g_hash_table_replace (data->last_modified_hash, g_strdup(dent->name), ctime);
04173             }
04174         }
04175 
04176         if (g_hash_table_size(data->current_file_id_hash) == 0) {
04177             *stop = TRUE;
04178             goto out;
04179         }
04180     }
04181 
04182     /* Files not found in the current commit are last modified in the previous
04183      * commit */
04184     g_hash_table_foreach_remove (data->current_file_id_hash,
04185                                  check_non_existing_files, data);
04186 
04187     if (g_hash_table_size(data->current_file_id_hash) == 0) {
04188         /* All files under this diretory have been calculated  */
04189         *stop = TRUE;
04190         goto out;
04191     }
04192 
04193 out:
04194     seaf_dir_free (dir);
04195 
04196     return ret;
04197 }
