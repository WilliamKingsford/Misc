00235 static int timeout_cb(CcnetProcessor *processor)
00236 {
00237     /* Since we set MAX_NUM_RETRY to 1, actually, there will be no retry. */
00238     if (processor->state == INIT) {
00239         if (++processor->num_retry < MAX_NUM_RETRY)
00240         {
00241             send_request(processor);
00242             return TRUE;
00243         } else {
00244             close_processor_in_timeout(processor);
00245             return FALSE;
00246         }
00247     }
00248 
00249     if (processor->state == WAIT_PUBKEY || processor->state == WAIT_CHALLENGE)
00250     {
00251         close_processor_in_timeout(processor);
00252         return FALSE;        
00253     }
00254 
00255     if (processor->state == WAIT_KEEPALIVE) {
00256         if (++processor->num_retry < MAX_NUM_RETRY) {
00257             send_keepalive (processor);
00258             return TRUE;
00259         } else {
00260             close_processor_in_timeout(processor);
00261             return FALSE;
00262         }
00263     }
00264 
00265     if (processor->state == FULL) {
00266         processor->num_retry = 0;
00267         send_keepalive(processor);
00268         return TRUE;
00269     }
00270 
00271     return FALSE;
00272 }
