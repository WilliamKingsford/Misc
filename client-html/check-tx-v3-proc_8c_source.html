00078 static char *
00079 encrypt_token (CcnetProcessor *processor, const char *token)
00080 {
00081     CcnetPeer *peer = NULL;
00082     char *enc_out = NULL;
00083     SeafileCrypt *crypt = NULL;
00084     unsigned char key[16], iv[16];

00086     char *output = NULL;
00087 
00088     if (!token)
00089         goto out;
00090 
00091     peer = ccnet_get_peer(seaf->ccnetrpc_client, processor->peer_id);
00092     if (!peer || !peer->session_key) {
00093         seaf_warning ("[check tx v3] peer or peer session key not exist\n");
00094         goto out;
00095     }
00096 
00097     EVP_BytesToKey (EVP_aes_128_cbc(), /* cipher mode */
00098                     EVP_sha1(),        /* message digest */
00099                     NULL,              /* slat */
00100                     (unsigned char*)peer->session_key,
00101                     strlen(peer->session_key),
00102                     1,   /* iteration times */
00103                     key, /* the derived key */
00104                     iv); /* IV, initial vector */
00105 
00106     crypt = seafile_crypt_new (1, key, iv);
00107     
00108     /* encrypt the token with session key, including the trailing null byte */
00109     if (seafile_encrypt (&enc_out, &len, token, strlen(token) + 1, crypt) < 0) {
00110         seaf_warning ("[check tx v3] failed to encrypt token\n");
00111         goto out;
00112     }
00113 
00114     output = g_malloc (len * 2 + 1);
00115     rawdata_to_hex ((unsigned char *)enc_out, output, len);
00116     output[len * 2] = '\0';
00117 
00118     
00119 out:
00120     g_free (crypt);
00121     g_free (enc_out);
00122     if (peer)
00123         g_object_unref(peer);
00124 
00125     return output;
00126 }
