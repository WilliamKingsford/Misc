01505 static int
01506 parse_mime_header (char *header, RecvFSM *fsm)
01507 {
01508     char *colon;
01509     char **params, **p;
01510 
01511     colon = strchr (header, ':');
01512     if (!colon) {
01513         seaf_warning ("[upload] bad mime header format.\n");
01514         return -1;
01515     }
01516 
01517     *colon = 0;
01518     if (strcmp (header, "Content-Disposition") == 0) {
01519         params = g_strsplit (colon + 1, ";", 0);
01520         for (p = params; *p != NULL; ++p)
01521             *p = g_strstrip (*p);
01522 
01523         if (!params || g_strv_length (params) < 2) {
01524             seaf_warning ("[upload] Too little params for mime header.\n");
01525             g_strfreev (params);
01526             return -1;
01527         }
01528         if (strcasecmp (params[0], "form-data") != 0) {
01529             seaf_warning ("[upload] Invalid Content-Disposition\n");
01530             g_strfreev (params);
01531             return -1;
01532         }
01533 
01534         for (p = params; *p != NULL; ++p) {
01535             if (strncasecmp (*p, "name", strlen("name")) == 0) {
01536                 fsm->input_name = get_mime_header_param_value (*p);
01537                 break;
01538             }
01539         }
01540         if (!fsm->input_name) {
01541             seaf_warning ("[upload] No input-name given.\n");
01542             g_strfreev (params);
01543             return -1;
01544         }
01545 
01546         if (strcmp (fsm->input_name, "file") == 0) {
01547             for (p = params; *p != NULL; ++p) {
01548                 if (strncasecmp (*p, "filename", strlen("filename")) == 0) {
01549                     fsm->file_name = get_mime_header_param_value (*p);
01550                     break;
01551                 }
01552             }
01553             if (!fsm->file_name) {
01554                 seaf_warning ("[upload] No filename given.\n");
01555                 g_strfreev (params);
01556                 return -1;
01557             }
01558         }
01559         g_strfreev (params);
01560     }
01561 
01562     return 0;
01563 }

01702 /*
01703    Example multipart form-data request content format:
01704 
01705    --AaB03x
01706    Content-Disposition: form-data; name="submit-name"
01707 
01708    Larry
01709    --AaB03x
01710    Content-Disposition: form-data; name="file"; filename="file1.txt"
01711    Content-Type: text/plain
01712 
01713    ... contents of file1.txt ...
01714    --AaB03x--
01715 */
01716 static evhtp_res
01717 upload_read_cb (evhtp_request_t *req, evbuf_t *buf, void *arg)
01718 {
01719     RecvFSM *fsm = arg;
01720     char *line;
01721     size_t len;
01722     gboolean no_line = FALSE;
01723     int res = EVHTP_RES_OK;
01724 
01725     if (fsm->state == RECV_ERROR)
01726         return EVHTP_RES_OK;
01727 
01728     /* Update upload progress. */
01729     if (fsm->progress) {
01730         fsm->progress->uploaded += (gint64)evbuffer_get_length(buf);
01731 
01732         seaf_debug ("progress: %lld/%lld\n",
01733                     fsm->progress->uploaded, fsm->progress->size);
01734     }
01735 
01736     evbuffer_add_buffer (fsm->line, buf);
01737     /* Drain the buffer so that evhtp don't copy it to another buffer
01738      * after this callback returns. 
01739      */
01740     evbuffer_drain (buf, evbuffer_get_length (buf));
01741 
01742     while (!no_line) {
01743         switch (fsm->state) {
01744         case RECV_INIT:
01745             line = evbuffer_readln (fsm->line, &len, EVBUFFER_EOL_CRLF_STRICT);
01746             if (line != NULL) {
01747                 seaf_debug ("[upload] boundary line: %s.\n", line);
01748                 if (!strstr (line, fsm->boundary)) {
01749                     seaf_warning ("[upload] no boundary found in the first line.\n");
01750                     free (line);
01751                     res = EVHTP_RES_BADREQ;
01752                     goto out;
01753                 } else {
01754                     fsm->state = RECV_HEADERS;
01755                     free (line);
01756                 }
01757             } else {
01758                 no_line = TRUE;
01759             }
01760             break;
01761         case RECV_HEADERS:
01762             line = evbuffer_readln (fsm->line, &len, EVBUFFER_EOL_CRLF_STRICT);
01763             if (line != NULL) {
01764                 seaf_debug ("[upload] mime header line: %s.\n", line);
01765                 if (len == 0) {
01766                     /* Read an blank line, headers end. */
01767                     free (line);
01768                     if (g_strcmp0 (fsm->input_name, "file") == 0) {
01769                         if (open_temp_file (fsm) < 0) {
01770                             seaf_warning ("[upload] Failed open temp file.\n");
01771                             res = EVHTP_RES_SERVERR;
01772                             goto out;
01773                         }
01774                     }
01775                     seaf_debug ("[upload] Start to recv %s.\n", fsm->input_name);
01776                     fsm->state = RECV_CONTENT;
01777                 } else if (parse_mime_header (line, fsm) < 0) {
01778                     free (line);
01779                     res = EVHTP_RES_BADREQ;
01780                     goto out;
01781                 } else {
01782                     free (line);
01783                 }
01784             } else {
01785                 no_line = TRUE;
01786             }
01787             break;
01788         case RECV_CONTENT:
01789             if (g_strcmp0 (fsm->input_name, "file") == 0)
01790                 res = recv_file_data (fsm, &no_line);
01791             else
