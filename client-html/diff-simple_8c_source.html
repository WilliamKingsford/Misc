00666 /* This function only resolve "strict" rename, i.e. two files must be
00667  * exactly the same.
00668  * Don't detect rename of empty files and empty dirs.
00669  */
00670 void
00671 diff_resolve_renames (GList **diff_entries)
00672 {
00673     GHashTable *deleted;
00674     GList *p;
00675     GList *added = NULL;
00676     DiffEntry *de;
00677     unsigned char empty_sha1[20];
00678 
00679     memset (empty_sha1, 0, 20);
00680 
00681     /* Hash and equal functions for raw sha1. */
00682     deleted = g_hash_table_new (ccnet_sha1_hash, ccnet_sha1_equal);
00683 
00684     /* Collect all "deleted" entries. */
00685     for (p = *diff_entries; p != NULL; p = p->next) {
00686         de = p->data;
00687         if ((de->status == DIFF_STATUS_DELETED ||
00688              de->status == DIFF_STATUS_DIR_DELETED) &&
00689             memcmp (de->sha1, empty_sha1, 20) != 0)
00690             g_hash_table_insert (deleted, de->sha1, p);
00691     }
00692 
00693     /* Collect all "added" entries into a separate list. */
00694     for (p = *diff_entries; p != NULL; p = p->next) {
00695         de = p->data;
00696         if ((de->status == DIFF_STATUS_ADDED ||
00697              de->status == DIFF_STATUS_DIR_ADDED) &&
00698             memcmp (de->sha1, empty_sha1, 20) != 0)
00699             added = g_list_prepend (added, p);
00700     }
00701 
00702     /* For each "added" entry, if we find a "deleted" entry with
00703      * the same content, we find a rename pair.
00704      */
00705     p = added;
00706     while (p != NULL) {
00707         GList *p_add, *p_del;
00708         DiffEntry *de_add, *de_del, *de_rename;
00709         int rename_status;
00710 
00711         p_add = p->data;
00712         de_add = p_add->data;
00713 
00714         p_del = g_hash_table_lookup (deleted, de_add->sha1);
00715         if (p_del) {
00716             de_del = p_del->data;
00717 
00718             if (de_add->status == DIFF_STATUS_DIR_ADDED)
00719                 rename_status = DIFF_STATUS_DIR_RENAMED;
00720             else
00721                 rename_status = DIFF_STATUS_RENAMED;
00722 
00723             de_rename = diff_entry_new (de_del->type, rename_status, 
00724                                         de_del->sha1, de_del->name);
00725             de_rename->new_name = g_strdup(de_add->name);
00726 
00727             *diff_entries = g_list_delete_link (*diff_entries, p_add);
00728             *diff_entries = g_list_delete_link (*diff_entries, p_del);
00729             *diff_entries = g_list_prepend (*diff_entries, de_rename);
00730 
00731             g_hash_table_remove (deleted, de_add->sha1);
00732 
00733             diff_entry_free (de_add);
00734             diff_entry_free (de_del);
00735         }
00736 
00737         p = g_list_delete_link (p, p);
00738     }
00739 
00740     g_hash_table_destroy (deleted);
00741 }

00743 static gboolean
00744 is_redundant_empty_dir (DiffEntry *de_dir, DiffEntry *de_file)
00745 {
00746     int dir_len;
00747 
00748     if (de_dir->status == DIFF_STATUS_DIR_ADDED &&
00749         de_file->status == DIFF_STATUS_DELETED)

00751         dir_len = strlen (de_dir->name);
00752         if (strlen (de_file->name) > dir_len &&
00753             strncmp (de_dir->name, de_file->name, dir_len) == 0)
00754             return TRUE;
00755     }
00756 
00757     if (de_dir->status == DIFF_STATUS_DIR_DELETED &&
00758         de_file->status == DIFF_STATUS_ADDED)
00759     {
00760         dir_len = strlen (de_dir->name);
00761         if (strlen (de_file->name) > dir_len &&
00762             strncmp (de_dir->name, de_file->name, dir_len) == 0)
00763             return TRUE;
00764     }
00765 
00766     return FALSE;

00769 /*
00770  * An empty dir entry may be added by deleting all the files under it.
00771  * Similarly, an empty dir entry may be deleted by adding some file in it.
00772  * In both cases, we don't want to include the empty dir entry in the
00773  * diff results.
00774  */
00775 void
00776 diff_resolve_empty_dirs (GList **diff_entries)
00777 {
00778     GList *empty_dirs = NULL;
00779     GList *p, *dir, *file;
00780     DiffEntry *de, *de_dir, *de_file;
00781 
00782     for (p = *diff_entries; p != NULL; p = p->next) {
00783         de = p->data;
00784         if (de->status == DIFF_STATUS_DIR_ADDED ||
00785             de->status == DIFF_STATUS_DIR_DELETED)
00786             empty_dirs = g_list_prepend (empty_dirs, p);
00787     }
00788 
00789     for (dir = empty_dirs; dir != NULL; dir = dir->next) {
00790         de_dir = ((GList *)dir->data)->data;
00791         for (file = *diff_entries; file != NULL; file = file->next) {
00792             de_file = file->data;
00793             if (is_redundant_empty_dir (de_dir, de_file)) {
00794                 *diff_entries = g_list_delete_link (*diff_entries, dir->data);
00795                 break;
00796             }
00797         }
00798     }
00799 
00800     g_list_free (empty_dirs);
00801 }
