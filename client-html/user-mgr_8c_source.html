00666

00669     SHA256_Update (&s, salt, sizeof(salt));

00674 static void
00675 hash_password_pbkdf2_sha256 (const char *passwd,
00676                              int iterations,
00677                              char **db_passwd)
00678 {
00679     guint8 sha[SHA256_DIGEST_LENGTH];
00680     guint8 salt[SHA256_DIGEST_LENGTH];
00681     char hashed_passwd[SHA256_DIGEST_LENGTH*2+1];
00682     char salt_str[SHA256_DIGEST_LENGTH*2+1];
00683 
00684     if (!RAND_bytes (salt, sizeof(salt))) {
00685         ccnet_warning ("Failed to generate salt "
00686                        "with RAND_bytes(), use RAND_pseudo_bytes().\n");
00687         RAND_pseudo_bytes (salt, sizeof(salt));
00688     }
00689 
00690     PKCS5_PBKDF2_HMAC (passwd, strlen(passwd),
00691                        salt, sizeof(salt),
00692                        iterations,
00693                        EVP_sha256(),
00694                        sizeof(sha), sha);
00695 
00696     rawdata_to_hex (sha, hashed_passwd, SHA256_DIGEST_LENGTH);
00697 
00698     rawdata_to_hex (salt, salt_str, SHA256_DIGEST_LENGTH);
00699 
00700     /* Encode password hash related information into one string, similar to Django. */
00701     GString *buf = g_string_new (NULL);
00702     g_string_printf (buf, "PBKDF2SHA256$%d$%s$%s",
00703                      iterations, salt_str, hashed_passwd);
00704     *db_passwd = g_string_free (buf, FALSE);
00705 }

00707 static gboolean
00708 validate_passwd_pbkdf2_sha256 (const char *passwd, const char *db_passwd)
00709 {
00710     char **tokens;
00711     char *salt_str, *hash;
00712     int iter;
00713     guint8 sha[SHA256_DIGEST_LENGTH];
00714     guint8 salt[SHA256_DIGEST_LENGTH];
00715     char hashed_passwd[SHA256_DIGEST_LENGTH*2+1];
00716 
00717     tokens = g_strsplit (db_passwd, "$", -1);
00718     if (!tokens || g_strv_length (tokens) != 4) {
00719         ccnet_warning ("Invalide db passwd format %s.\n", db_passwd);
00720         return FALSE;
00721     }
00722 
00723     iter = atoi (tokens[1]);
00724     salt_str = tokens[2];
00725     hash = tokens[3];
00726 
00727     hex_to_rawdata (salt_str, salt, SHA256_DIGEST_LENGTH);
00728 
00729     PKCS5_PBKDF2_HMAC (passwd, strlen(passwd),
00730                        salt, sizeof(salt),
00731                        iter,
00732                        EVP_sha256(),
00733                        sizeof(sha), sha);
00734     rawdata_to_hex (sha, hashed_passwd, SHA256_DIGEST_LENGTH);
00735 
00736     gboolean ret = (strcmp (hash, hashed_passwd) == 0);
00737 
00738     g_strfreev (tokens);
00739     return ret;
00740 }

00742 static gboolean
00743 validate_passwd (const char *passwd, const char *stored_passwd,
00744                  gboolean *need_upgrade)
00745 {
00746     char hashed_passwd[SHA256_DIGEST_LENGTH * 2 + 1];
00747     int hash_len = strlen(stored_passwd);
00748 
00749     *need_upgrade = FALSE;
00750 
00751     if (hash_len == SHA256_DIGEST_LENGTH * 2) {
00752         hash_password_salted (passwd, hashed_passwd);
00753         *need_upgrade = TRUE;
00754     } else if (hash_len == SHA_DIGEST_LENGTH * 2) {
00755         hash_password (passwd, hashed_passwd);
00756         *need_upgrade = TRUE;
00757     } else {
00758         return validate_passwd_pbkdf2_sha256 (passwd, stored_passwd);
00759     }
00760 
00761     if (strcmp (hashed_passwd, stored_passwd) == 0)
00762         return TRUE;
00763     else
00764         return FALSE;
00765 }
