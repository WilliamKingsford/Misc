00685                  * without updating worktree. */
00686                 update_file_flags (o, a_sha, a_mode, a_modifier, a_mtime, path, 1, 0);
00687         }

00690         /* case D: Modified in both, but differently. */
00691         if (memcmp(a_sha, b_sha, 20) != 0 || a_mode != b_mode) {

00695

00697                 new_path = gen_conflict_path_wrapper (o->repo_id, o->version,
00698                                                       o->remote_head, path,
00699                                                       path);
00700                 if (!new_path)

00702                                                  o->branch2,
00703                                                  (gint64)time(NULL));

00705 
00706             /* Dont update index. */
00707             /* Keep my version, rename other's version. */

00710         } else {

00712         }
00713     } else if (!o_sha && !a_sha && !b_sha) {
00714         /*

00716          * we had that path and want to actively remove it.
00717          */
00718         remove_file(o, 1, path, !a_mode, current_mtime);
00719     } else
00720         g_error("Fatal merge failure, shouldn't happen.");

00725 static int is_garbage_empty_dir (struct index_state *index, const char *name)
00726 {
00727     int pos = index_name_pos (index, name, strlen(name));
00728 
00729     /*
00730      * If pos >= 0, ++pos to the next entry in the index.
00731      * If pos < 0, -pos = (the position this entry *should* be) + 1.
00732      * So -pos-1 is the first entry larger than this entry.
00733      */
00734     if (pos >= 0)
00735         pos++;
00736     else
00737         pos = -pos-1;
00738 
00739     struct cache_entry *next;
00740     int this_len = strlen (name);
00741     while (pos < index->cache_nr) {
00742         next = index->cache[pos];
00743 
00744         /* If 'name' is the prefix of next->name but they are unequal,
00745          * it means there are entries under this empty dir. So this "emtpy dir"
00746          * is useless.
00747          */
00748         if (strncmp (name, next->name, this_len) != 0)
00749             break;
00750         if (strcmp (name, next->name) != 0)
00751             return 1;
00752         ++pos;
00753     }
00754 
00755     return 0;
00756 }

00758 /*
00759  * per entry merge function for D/F (and/or rename) conflicts.  In the
00760  * cases we can cleanly resolve D/F conflicts, process_entry() can
00761  * clean out all the files below the directory for us.  All D/F
00762  * conflict cases must be handled here at the end to make sure any
00763  * directories that can be cleaned out, are.
00764  */
00765 static int process_df_entry(struct merge_options *o,
00766                             const char *path,
00767                             struct stage_data *entry)
00768 {
00769     int clean_merge = 1;
00770     unsigned o_mode = entry->stages[1].mode;
00771     unsigned a_mode = entry->stages[2].mode;
00772     unsigned b_mode = entry->stages[3].mode;
00773     unsigned char *o_sha = o_mode ? entry->stages[1].sha : NULL;
00774     unsigned char *a_sha = a_mode ? entry->stages[2].sha : NULL;
00775     unsigned char *b_sha = b_mode ? entry->stages[3].sha : NULL;
00776     guint64 a_mtime = entry->stages[2].mtime;
00777     guint64 b_mtime = entry->stages[3].mtime;
00778     char *a_modifier = entry->stages[2].modifier;
00779     char *b_modifier = entry->stages[3].modifier;
00780     SeafStat st;
00781     char *real_path = g_build_path(PATH_SEPERATOR, o->worktree, path, NULL);
00782     char *new_path = NULL;
00783 
00784     entry->processed = 1;
00785     if (o_sha && (!a_sha || !b_sha)) {
00786         /* Modify/delete; deleted side may have put a directory in the way */
00787         if (b_sha) {
00788             if (seaf_stat (real_path, &st) == 0 && S_ISDIR(st.st_mode)) {
00789                 /* D/F conflict. */
00790                 clean_merge = 0;
00791 
00792                 if (!o->collect_blocks_only) {
00793                     new_path = gen_conflict_path_wrapper (o->repo_id, o->version,
00794                                                           o->remote_head, path,
00795                                                           path);
00796                     if (!new_path)
00797                         new_path = gen_conflict_path(path,
00798                                                      o->branch2,
00799                                                      (gint64)time(NULL));
00800                 }
00801 
00802                 update_file(o, 0, b_sha, b_mode, b_modifier, b_mtime, new_path);
00803                 g_free (new_path);
00804             } else {
00805                 /* Modify/Delete conflict. Don't consider as unclean. */
00806                 clean_merge = update_file(o, 1, b_sha, b_mode, b_modifier, b_mtime, path);
00807             }
00808         } else {
00809             if (seaf_stat (real_path, &st) == 0 && S_ISDIR(st.st_mode)) {
00810                 clean_merge = 0;
00811             } else {
00812                 /* Clean merge. Just need to update index. */
00813                 update_file_flags (o, a_sha, a_mode, a_modifier, a_mtime, path, 1, 0);
00814             }
00815         }
00816     } else if (!o_sha && !!a_sha != !!b_sha) {
00817         unsigned char *sha = a_sha ? a_sha : b_sha;
00818         unsigned mode = a_sha ? a_mode : b_mode;
00819         char *modifier = a_sha ? a_modifier : b_modifier;
00820         guint64 mtime = a_sha ? a_mtime : b_mtime;
00821 
00822         /* directory -> (directory, empty dir) or
00823          * directory -> (empty dir, directory) */
00824         if (S_ISDIR(mode)) {
00825             /* Merge is always clean. If the merge result is non empty dir,
00826              * remove the empty dir entry from the index.
00827              */
00828             if (is_garbage_empty_dir (o->index, path))
00829                 remove_file_from_index (o->index, path);
00830             else if (a_sha)
00831                 update_file_flags (o, sha, mode, modifier, mtime, path, 1, 0);
00832             else
00833                 update_file_flags (o, sha, mode, modifier, mtime, path, 1, 1);
00834             goto out;
00835         }
00836 
00837         /* directory -> (directory, file) */
00838         if (b_sha) {
00839             if (seaf_stat (real_path, &st) == 0 && S_ISDIR(st.st_mode)) {
00840                 /* D/F conflict. */
00841                 clean_merge = 0;
00842 
00843                 if (!o->collect_blocks_only) {
00844                     new_path = gen_conflict_path_wrapper (o->repo_id, o->version,
00845                                                           o->remote_head, path,
00846                                                           path);
00847                     if (!new_path)
00848                         new_path = gen_conflict_path(path,
00849                                                      o->branch2,
00850                                                      (gint64)time(NULL));
00851                 }
00852 
00853                 update_file(o, 0, b_sha, b_mode, b_modifier, b_mtime, new_path);
00854                 g_free (new_path);
00855             } else {
00856                 /* Clean merge. */
00857                 clean_merge = update_file(o, 1, b_sha, b_mode, b_modifier, b_mtime, path);
00858             }
00859         } else {
00860             if (seaf_stat (real_path, &st) == 0 && S_ISDIR(st.st_mode)) {
00861                 clean_merge = 0;
00862             } else {
00863                 /* Clean merge. Just need to update index. */
00864                 update_file_flags (o, a_sha, a_mode, a_modifier, a_mtime, path, 1, 0);
00865             }
00866         }
00867     } else {
00868         entry->processed = 0;

00950                     const char *ca_root,
