00194 /*
00195  * We only recognize two consecutive "moved" events with the same cookie as
00196  * a rename pair. The processing logic is:
00197  * 1. Receive a MOVED_FROM event, set last_cookie and old_path, set processing to TRUE
00198  * 2. If the next event is MOVED_TO, and with the same cookie, then add an
00199  *    WT_EVENT_RENAME event to the queue.
00200  * 3. Otherwise, recognize them as one delete event followed by one
00201  *    create event
00202  *
00203  * This is a two-state state machine. The states are 'not processing rename' and
00204  * 'processing rename'.
00205  */
00206 static void
00207 handle_rename (int in_fd,
00208                RepoWatchInfo *info,
00209                struct inotify_event *event,
00210                const char *worktree,
00211                const char *filename,
00212                gboolean last_event)
00213 {
00214     WTStatus *status = info->status;
00215     RenameInfo *rename_info = info->rename_info;
00216 
00217     if (event->mask & IN_MOVED_FROM)
00218         seaf_debug ("(%d) Move %s ->\n", event->cookie, event->name);
00219     else if (event->mask & IN_MOVED_TO)
00220         seaf_debug ("(%d) Move -> %s.\n", event->cookie, event->name);
00221 
00222     if (!rename_info->processing) {
00223         if (event->mask & IN_MOVED_FROM) {
00224             if (!last_event) {
00225                 set_rename_processing_state (rename_info, event->cookie, filename);
00226             } else {
00227                 /* Rename event pair should be in one batch of events.
00228                  * If a MOVED_FROM event is the last event in a batch,
00229                  * the path should be moved out of the repo.
00230                  */
00231                 add_event_to_queue (status, WT_EVENT_DELETE, filename, NULL);
00232             }
00233         } else if (event->mask & IN_MOVED_TO) {
00234             /* A file/dir was moved into this repo. */
00235             /* Add watch and produce events. */
00236             add_event_to_queue (status, WT_EVENT_CREATE_OR_UPDATE, filename, NULL);
00237             add_watch_recursive (info, in_fd, worktree, filename, FALSE);
00238         }
00239     } else {
00240         if (event->mask & IN_MOVED_FROM) {
00241             /* A file/dir was moved out of this repo.
00242              * Output the last MOVED_FROM event as DELETE event
00243              */
00244             add_event_to_queue (status, WT_EVENT_DELETE, rename_info->old_path, NULL);
00245 
00246             if (!last_event) {
00247                 /* Stay in processing state. */
00248                 rename_info->last_cookie = event->cookie;
00249                 g_free (rename_info->old_path);
00250                 rename_info->old_path = g_strdup(filename);
00251             } else {
00252                 /* Another file/dir was moved out of this repo. */
00253                 add_event_to_queue (status, WT_EVENT_DELETE, filename, NULL);
00254                 unset_rename_processing_state (rename_info);
00255             }
00256         } else if (event->mask & IN_MOVED_TO) {
00257             if (event->cookie == rename_info->last_cookie) {
00258                 /* Rename pair detected. */
00259                 add_event_to_queue (status, WT_EVENT_RENAME,
00260                                     rename_info->old_path, filename);
00261             } else {
00262                 /* A file/dir was moved out of the repo, followed by
00263                  * aother file/dir was moved into this repo.
00264                  */
00265                 add_event_to_queue (status, WT_EVENT_DELETE,
00266                                     rename_info->old_path, NULL);
00267                 add_event_to_queue (status, WT_EVENT_CREATE_OR_UPDATE,
00268                                     filename, NULL);
00269             }
00270             /* Need to update wd -> path mapping. */
00271             add_watch_recursive (info, in_fd, worktree, filename, FALSE);
00272             unset_rename_processing_state (rename_info);
00273         } else {
00274             /* A file/dir was moved out of this repo, followed by another
00275              * file operations.
00276              */
00277             add_event_to_queue (status, WT_EVENT_DELETE, rename_info->old_path, NULL);
00278             unset_rename_processing_state (rename_info);
00279         }
00280     }

00319     char *filename;
00320     gboolean update_last_changed = TRUE;
00321     gboolean add_to_queue = TRUE;
00322 
00323     /* An inotfy watch has been removed, we don't care about this for now. */
00324     if ((event->mask & IN_IGNORED) || (event->mask & IN_UNMOUNT))
00325         return;
00326 
00327     /* Kernel event queue was overflowed, some events may lost. */
00328     if (event->mask & IN_Q_OVERFLOW) {
00329         add_event_to_queue (status, WT_EVENT_OVERFLOW, NULL, NULL);
00330         return;
00331     }
00332 
00333     if (handle_consecutive_duplicate_event (info, event))
00334         add_to_queue = FALSE;
00335 
00336     filename = g_build_filename (parent, event->name, NULL);
00337 
00338     handle_rename (in_fd, info, event, worktree, filename, last_event);
00339 
00340     if (event->mask & IN_MODIFY) {
00341         seaf_debug ("Modified %s.\n", filename);
00342         if (add_to_queue)
00343             add_event_to_queue (status, WT_EVENT_CREATE_OR_UPDATE, filename, NULL);
00344     } else if (event->mask & IN_CREATE) {
00345         seaf_debug ("Created %s.\n", filename);
00346 
00347         /* Nautilus's file copy operation doesn't trigger write events.
00348          * If the user copy a large file into the repo, only a create
00349          * event and a close_write event will be received. If we process
00350          * the create event, we'll certainly try to index a file when it's
00351          * still being copied. So we'll ignore create event for files.
00352          * Since write and close_write events will always be triggered,
00353          * we don't need to worry about missing this file.
00354          */
00355         char *fullpath = g_build_filename (worktree, filename, NULL);
00356         struct stat st;
00357         if (lstat (fullpath, &st) < 0 ||
00358             (!S_ISDIR(st.st_mode) && !S_ISLNK(st.st_mode))) {
00359             g_free (fullpath);
00360             update_last_changed = FALSE;
00361             goto out;
00362         }
00363         g_free (fullpath);
00364 
00365         /* We now know it's a directory or a symlink. */
00366 
00367         /* Files or dirs could have been added under this dir before we
00368          * watch it. So it's safer to scan this dir. At most time we don't
00369          * have to scan recursively and very few new files will be found.
00370          */
00371         add_event_to_queue (status, WT_EVENT_CREATE_OR_UPDATE, filename, NULL);
00372         add_watch_recursive (info, in_fd, worktree, filename, FALSE);
00373     } else if (event->mask & IN_DELETE) {
00374         seaf_debug ("Deleted %s.\n", filename);
00375         add_event_to_queue (status, WT_EVENT_DELETE, filename, NULL);
00376     } else if (event->mask & IN_CLOSE_WRITE) {
00377         seaf_debug ("Close write %s.\n", filename);
00378         if (add_to_queue)
00379             add_event_to_queue (status, WT_EVENT_CREATE_OR_UPDATE, filename, NULL);
00380     } else if (event->mask & IN_ATTRIB) {
00381         seaf_debug ("Attribute changed %s.\n", filename);
00382         add_event_to_queue (status, WT_EVENT_ATTRIB, filename, NULL);
00383     }
00384 
00385 out:
00386     g_free (filename);
00387     if (update_last_changed)
00388         g_atomic_int_set (&info->status->last_changed, (gint)time(NULL));
