00195     found = 0;

00197         for (q = name; *q == *p++; q++)
00198             if (*q == 0)                /* exact match? */
00199                 return c;
00200         if (!*q) {      /* the name was a prefix */
00201             if (q - name > longest) {
00202                 longest = q - name;
00203                 nmatches = 1;

00206                 nmatches++;

00208     }

00211         return (struct cmd *)-1;
