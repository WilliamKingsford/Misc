00714 static int
00715 fast_forward_or_merge (const char *repo_id,
00716                        SeafCommit *base,
00717                        SeafCommit *new_commit)
00718 {
00719 #define MAX_RETRY_COUNT 3
00720 
00721     SeafRepo *repo = NULL;
00722     SeafCommit *current_head = NULL, *merged_commit = NULL;
00723     int retry_cnt = 0;
00724     int ret = 0;
00725 
00726     repo = seaf_repo_manager_get_repo (seaf->repo_mgr, repo_id);
00727     if (!repo) {
00728         seaf_warning ("Repo %s doesn't exist.\n", repo_id);
00729         ret = -1;
00730         goto out;
00731     }
00732 
00733 retry:
00734     current_head = seaf_commit_manager_get_commit (seaf->commit_mgr,
00735                                                    repo->id, repo->version,
00736                                                    repo->head->commit_id);
00737     if (!current_head) {
00738         seaf_warning ("Failed to find head commit of %s.\n", repo_id);
00739         ret = -1;
00740         goto out;
00741     }
00742 
00743     /* Merge if base and head are not the same. */
00744     if (strcmp (base->commit_id, current_head->commit_id) != 0) {
00745         MergeOptions opt;
00746         const char *roots[3];
00747         char *desc = NULL;
00748 
00749         memset (&opt, 0, sizeof(opt));
00750         opt.n_ways = 3;
00751         memcpy (opt.remote_repo_id, repo_id, 36);
00752         memcpy (opt.remote_head, new_commit->commit_id, 40);
00753         opt.do_merge = TRUE;
00754 
00755         roots[0] = base->root_id; /* base */
00756         roots[1] = current_head->root_id; /* head */
00757         roots[2] = new_commit->root_id;      /* remote */
00758 
00759         if (seaf_merge_trees (repo->store_id, repo->version, 3, roots, &opt) < 0) {
00760             seaf_warning ("Failed to merge.\n");
00761             ret = -1;
00762             goto out;
00763         }
00764 
00765         if (!opt.conflict)
00766             desc = g_strdup("Auto merge by system");
00767         else {
00768             desc = gen_merge_description (repo,
00769                                           opt.merged_tree_root,
00770                                           current_head->root_id,
00771                                           new_commit->root_id);
00772             if (!desc)
00773                 desc = g_strdup("Auto merge by system");
00774         }
00775 
00776         merged_commit = seaf_commit_new(NULL, repo->id, opt.merged_tree_root,
00777                                         new_commit->creator_name, EMPTY_SHA1,
00778                                         desc,
00779                                         0);
00780         g_free (desc);
00781 
00782         merged_commit->parent_id = g_strdup (current_head->commit_id);
00783         merged_commit->second_parent_id = g_strdup (new_commit->commit_id);
00784         merged_commit->new_merge = TRUE;
00785         if (opt.conflict)
00786             merged_commit->conflict = TRUE;
00787         seaf_repo_to_commit (repo, merged_commit);
00788 
00789         if (seaf_commit_manager_add_commit (seaf->commit_mgr, merged_commit) < 0) {
00790             seaf_warning ("Failed to add commit.\n");
00791             ret = -1;
00792             goto out;
00793         }
00794     } else {
00795         seaf_commit_ref (new_commit);
00796         merged_commit = new_commit;
00797     }
00798 
00799     seaf_branch_set_commit(repo->head, merged_commit->commit_id);
00800 
00801     if (seaf_branch_manager_test_and_update_branch(seaf->branch_mgr,
00802                                                    repo->head,
00803                                                    current_head->commit_id) < 0)
00804     {
00805         seaf_repo_unref (repo);
00806         repo = NULL;
00807         seaf_commit_unref (current_head);
00808         current_head = NULL;
00809         seaf_commit_unref (merged_commit);
00810         merged_commit = NULL;
00811 
00812         repo = seaf_repo_manager_get_repo (seaf->repo_mgr, repo_id);
00813         if (!repo) {
00814             seaf_warning ("Repo %s doesn't exist.\n", repo_id);
00815             ret = -1;
00816             goto out;
00817         }
00818 
00819         if (++retry_cnt <= MAX_RETRY_COUNT) {
00820             seaf_message ("Concurrent branch update, retry.\n");
00821             /* Sleep random time between 100 and 1000 millisecs. */
00822             USLEEP (g_random_int_range(1, 11) * 100 * 1000);
00823             goto retry;
00824         } else {
00825             seaf_warning ("Stop retrying.\n");
00826             ret = -1;
00827             goto out;
00828         }
00829     }
00830 
00831 out:

00833     seaf_commit_unref (merged_commit);
