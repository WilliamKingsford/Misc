00712 static int
00713 fast_forward_or_merge (const char *repo_id,
00714                        SeafCommit *base,
00715                        SeafCommit *new_commit)
00716 {
00717 #define MAX_RETRY_COUNT 3
00718 
00719     SeafRepo *repo = NULL;
00720     SeafCommit *current_head = NULL, *merged_commit = NULL;
00721     int retry_cnt = 0;
00722     int ret = 0;
00723 
00724     repo = seaf_repo_manager_get_repo (seaf->repo_mgr, repo_id);
00725     if (!repo) {
00726         seaf_warning ("Repo %s doesn't exist.\n", repo_id);
00727         ret = -1;
00728         goto out;
00729     }
00730 
00731 retry:
00732     current_head = seaf_commit_manager_get_commit (seaf->commit_mgr,
00733                                                    repo->id, repo->version,
00734                                                    repo->head->commit_id);
00735     if (!current_head) {
00736         seaf_warning ("Failed to find head commit of %s.\n", repo_id);
00737         ret = -1;
00738         goto out;
00739     }
00740 
00741     /* Merge if base and head are not the same. */
00742     if (strcmp (base->commit_id, current_head->commit_id) != 0) {
00743         MergeOptions opt;
00744         const char *roots[3];
00745         char *desc = NULL;
00746 
00747         memset (&opt, 0, sizeof(opt));
00748         opt.n_ways = 3;
00749         memcpy (opt.remote_repo_id, repo_id, 36);
00750         memcpy (opt.remote_head, new_commit->commit_id, 40);
00751         opt.do_merge = TRUE;
00752 
00753         roots[0] = base->root_id; /* base */
00754         roots[1] = current_head->root_id; /* head */
00755         roots[2] = new_commit->root_id;      /* remote */
00756 
00757         if (seaf_merge_trees (repo->store_id, repo->version, 3, roots, &opt) < 0) {
00758             seaf_warning ("Failed to merge.\n");
00759             ret = -1;
00760             goto out;
00761         }
00762 
00763         if (!opt.conflict)
00764             desc = g_strdup("Auto merge by system");
00765         else {
00766             desc = gen_merge_description (repo,
00767                                           opt.merged_tree_root,
00768                                           current_head->root_id,
00769                                           new_commit->root_id);
00770             if (!desc)
00771                 desc = g_strdup("Auto merge by system");
00772         }
00773 
00774         merged_commit = seaf_commit_new(NULL, repo->id, opt.merged_tree_root,
00775                                         new_commit->creator_name, EMPTY_SHA1,
00776                                         desc,
00777                                         0);
00778         g_free (desc);
00779 
00780         merged_commit->parent_id = g_strdup (current_head->commit_id);
00781         merged_commit->second_parent_id = g_strdup (new_commit->commit_id);
00782         merged_commit->new_merge = TRUE;
00783         if (opt.conflict)
00784             merged_commit->conflict = TRUE;
00785         seaf_repo_to_commit (repo, merged_commit);
00786 
00787         if (seaf_commit_manager_add_commit (seaf->commit_mgr, merged_commit) < 0) {
00788             seaf_warning ("Failed to add commit.\n");
00789             ret = -1;
00790             goto out;
00791         }
00792     } else {
00793         seaf_commit_ref (new_commit);
00794         merged_commit = new_commit;
00795     }
00796 
00797     seaf_branch_set_commit(repo->head, merged_commit->commit_id);
00798 
00799     if (seaf_branch_manager_test_and_update_branch(seaf->branch_mgr,
00800                                                    repo->head,
00801                                                    current_head->commit_id) < 0)
00802     {
00803         seaf_repo_unref (repo);
00804         repo = NULL;
00805         seaf_commit_unref (current_head);
00806         current_head = NULL;
00807         seaf_commit_unref (merged_commit);
00808         merged_commit = NULL;
00809 
00810         repo = seaf_repo_manager_get_repo (seaf->repo_mgr, repo_id);
00811         if (!repo) {
00812             seaf_warning ("Repo %s doesn't exist.\n", repo_id);
00813             ret = -1;
00814             goto out;
00815         }
00816 
00817         if (++retry_cnt <= MAX_RETRY_COUNT) {
00818             seaf_message ("Concurrent branch update, retry.\n");
00819             /* Sleep random time between 100 and 1000 millisecs. */
00820             USLEEP (g_random_int_range(1, 11) * 100 * 1000);
00821             goto retry;
00822         } else {
00823             seaf_warning ("Stop retrying.\n");
00824             ret = -1;
00825             goto out;
00826         }
00827     }
00828 
00829 out:

00831     seaf_commit_unref (merged_commit);
