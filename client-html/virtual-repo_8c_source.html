00889 static int
00890 schedule_merge_tasks (void *vscheduler)
00891 {
00892     MergeScheduler *scheduler = vscheduler;
00893     int n_running = g_hash_table_size (scheduler->running);
00894     MergeTask *task;
00895 
00896     /* seaf_debug ("Waiting tasks %d, running tasks %d.\n", */
00897     /*             g_queue_get_length (scheduler->queue), n_running); */
00898 
00899     if (n_running >= MAX_RUNNING_TASKS)
00900         return TRUE;
00901 
00902     pthread_mutex_lock (&scheduler->q_lock);
00903 
00904     while (n_running < MAX_RUNNING_TASKS) {
00905         task = g_queue_pop_head (scheduler->queue);
00906         if (!task)
00907             break;
00908 
00909         if (!g_hash_table_lookup (scheduler->running, task->repo_id)) {
00910             int ret = ccnet_job_manager_schedule_job (scheduler->tpool,
00911                                                       merge_virtual_repo,
00912                                                       merge_virtual_repo_done,
00913                                                       task);
00914             if (ret < 0) {
00915                 g_queue_push_tail (scheduler->queue, task);
00916                 break;
00917             }
00918 
00919             g_hash_table_insert (scheduler->running,
00920                                  g_strdup(task->repo_id),
00921                                  task);
00922             n_running++;
00923 
00924             seaf_debug ("Run task for repo %.8s.\n", task->repo_id);
00925         } else {
00926             seaf_debug ("A task for repo %.8s is already running.\n", task->repo_id);
00927 
00928             g_queue_push_tail (scheduler->queue, task);
00929             break;
00930         }
00931     }
00932 
00933     pthread_mutex_unlock (&scheduler->q_lock);
00934 
00935     return TRUE;
00936 }
