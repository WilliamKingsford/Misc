00177 /* Traverse the commit graph until remote_id is met or a merged commit
00178  * (commit with two parents) is met.
00179  *
00180  * If a merged commit is met before remote_id, that implies that
00181  * we did a real merge when merged with the branch headed by remote_id.
00182  * In this case we'll need more computation to find out the "delta" commits
00183  * between these two branches. Otherwise, if the merge was a fast-forward
00184  * one, it's enough to just send all the commits between our head commit
00185  * and remote_id.
00186  */
00187 static gboolean

00189 {
00190     CcnetProcessor *processor = data;
00191     TransferTask *task = ((SeafileSendcommitV3Proc *)processor)->tx_task;
00192     USE_PRIV;
00193 
00194     if (priv->remote_id[0] != 0 &&
00195         strcmp (priv->remote_id, commit->commit_id) == 0) {
00196         *stop = TRUE;
00197         return TRUE;
00198     }
00199 
00200     if (commit->second_parent_id != NULL) {
00201         *stop = TRUE;

00204     }
