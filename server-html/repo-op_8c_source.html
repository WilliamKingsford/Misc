01490 copy_seafile (SeafRepo *src_repo, SeafRepo *dst_repo, const char *file_id,

01495     file = seaf_fs_manager_get_seafile (seaf->fs_mgr,
01496                                         src_repo->store_id, src_repo->version,
01497                                         file_id);

01499         seaf_warning ("Failed to get file object %s from repo %s.\n",
01500                       file_id, src_repo->id);

01503

01505     file->version = seafile_version_from_repo_version(dst_repo->version);

01548 static char *
01549 copy_recursive (SeafRepo *src_repo, SeafRepo *dst_repo,
01550                 const char *obj_id, guint32 mode, const char *modifier,
01551                 CopyTask *task, guint64 *size)
01552 {
01553     if (S_ISREG(mode)) {
01554         return copy_seafile (src_repo, dst_repo, obj_id, task, size);
01555     } else if (S_ISDIR(mode)) {
01556         SeafDir *src_dir = NULL, *dst_dir = NULL;
01557         GList *dst_ents = NULL, *ptr;
01558         char *new_id = NULL;
01559         SeafDirent *dent, *new_dent = NULL;
01560 
01561         src_dir = seaf_fs_manager_get_seafdir (seaf->fs_mgr,
01562                                                src_repo->store_id,
01563                                                src_repo->version,
01564                                                obj_id);

01566             seaf_warning ("Seafdir %s doesn't exist in repo %s.\n",
01567                           obj_id, src_repo->id);
01568             return NULL;
01569         }
01570 
01571         for (ptr = src_dir->entries; ptr; ptr = ptr->next) {
01572             dent = ptr->data;
01573 
01574             guint64 new_size = 0;
01575             new_id = copy_recursive (src_repo, dst_repo,
01576                                      dent->id, dent->mode, modifier, task, &new_size);
01577             if (!new_id) {

01579                 return NULL;
01580             }
01581 
01582             new_dent = seaf_dirent_new (dir_version_from_repo_version(dst_repo->version),
01583                                         new_id, dent->mode, dent->name,
01584                                         (gint64)time(NULL), modifier, new_size);
01585             dst_ents = g_list_prepend (dst_ents, new_dent);
01586             g_free (new_id);
01587         }
01588         dst_ents = g_list_reverse (dst_ents);
01589 
01590         seaf_dir_free (src_dir);
01591 
01592         dst_dir = seaf_dir_new (NULL, dst_ents,
01593                                 dir_version_from_repo_version(dst_repo->version));
01594         if (seaf_dir_save (seaf->fs_mgr,
01595                            dst_repo->store_id, dst_repo->version,
01596                            dst_dir) < 0) {
01597             seaf_warning ("Failed to save new dir.\n");
01598             seaf_dir_free (dst_dir);
01599             return NULL;
01600         }
01601 
01602         char *ret = g_strdup(dst_dir->dir_id);
01603         *size = 0;
01604         seaf_dir_free (dst_dir);
01605         return ret;
01606     }
01607 
01608     return NULL;

02393 static char *
02394 rename_file_recursive(SeafRepo *repo,
02395                       const char *dir_id,
02396                       const char *to_path,
02397                       const char *oldname,
02398                       const char *newname)
02399 {
02400     SeafDir *olddir, *newdir;
02401     SeafDirent *dent;
02402     GList *ptr;
02403     char *to_path_dup = NULL;
02404     char *remain = NULL;
02405     char *slash;
02406     char *id = NULL;
02407 
02408     olddir = seaf_fs_manager_get_seafdir_sorted(seaf->fs_mgr,
02409                                                 repo->store_id, repo->version,
02410                                                 dir_id);
02411     if (!olddir)
02412         return NULL;
02413 
02414     /* we reach the target dir. */
02415     if (*to_path == '\0') {
02416         SeafDirent *old, *newdent = NULL;
02417         GList *newentries = NULL, *p;
02418 
02419         /* When renameing, there is a pitfall: we can't simply rename the
02420          * dirent, since the dirents are required to be sorted in descending
02421          * order. We need to copy all old dirents except the target dirent,
02422          * and then rename the target dirent, and then insert the new
02423          * dirent, so that we can maintain the descending order of dirents. */
02424         for (p = olddir->entries; p != NULL; p = p->next) {
02425             old = p->data;
02426             if (strcmp(old->name, oldname) != 0) {
02427                 newentries = g_list_prepend (newentries, seaf_dirent_dup(old));
02428             } else {
02429                 newdent = seaf_dirent_new (old->version, old->id, old->mode,
02430                                            newname, old->mtime,
02431                                            old->modifier, old->size);
02432             }
02433         }
02434 
02435         newentries = g_list_reverse (newentries);
02436 
02437         if (newdent) {
02438             newentries = g_list_insert_sorted(newentries, newdent, compare_dirents);
02439         }
02440 
02441         newdir = seaf_dir_new (NULL, newentries,
02442                                dir_version_from_repo_version(repo->version));
02443         seaf_dir_save (seaf->fs_mgr, repo->store_id, repo->version, newdir);
02444         id = g_strndup (newdir->dir_id, 40);
02445         seaf_dir_free (newdir);
02446 
02447         goto out;
02448     }
02449 
02450     to_path_dup = g_strdup (to_path);
02451     slash = strchr (to_path_dup, '/');
02452 
02453     if (!slash) {
02454         remain = to_path_dup + strlen(to_path_dup);
02455     } else {
02456         *slash = '\0';
02457         remain = slash + 1;
02458     }
02459 
02460     for (ptr = olddir->entries; ptr; ptr = ptr->next) {
02461         dent = (SeafDirent *)ptr->data;
02462 
02463         if (strcmp(dent->name, to_path_dup) != 0)
02464             continue;
02465 
02466         id = rename_file_recursive (repo, dent->id, remain, oldname, newname);
02467         if (id != NULL) {
02468             memcpy(dent->id, id, 40);
02469             dent->id[40] = '\0';
02470         }
02471         break;
02472     }
02473     
02474     if (id != NULL) {
02475         /* Create a new SeafDir. */
02476         GList *new_entries;
02477         
02478         new_entries = dup_seafdir_entries (olddir->entries);
02479         newdir = seaf_dir_new (NULL, new_entries,
02480                                dir_version_from_repo_version(repo->version));
02481         seaf_dir_save (seaf->fs_mgr, repo->store_id, repo->version, newdir);
02482         
02483         g_free(id);
02484         id = g_strndup(newdir->dir_id, 40);
02485         seaf_dir_free (newdir);
02486     }
02487 
02488 out:
02489     g_free (to_path_dup);
02490     seaf_dir_free(olddir);
02491     return id;
02492 }

02584 static char *

02587                    const char *to_path,

02602         return NULL;
02603 
02604     /* we reach the target dir. Update the target dirent. */
02605     if (*to_path == '\0') {

02607         SeafDirent *dent;
02608

02610             dent = p->data;

02614                 newentries = g_list_prepend (newentries, seaf_dirent_dup(dent));
02615             }
02616         }
02617

02629     slash = strchr (to_path_dup, '/');
02630 
02631     if (!slash) {
02632         remain = to_path_dup + strlen(to_path_dup);
02633     } else {
02634         *slash = '\0';
02635         remain = slash + 1;
02636     }
02637

02639         dent = (SeafDirent *)ptr->data;

02645         if (id != NULL) {
02646             memcpy(dent->id, id, 40);
02647             dent->id[40] = '\0';

02660                                dir_version_from_repo_version(repo->version));
02661         seaf_dir_save (seaf->fs_mgr, repo->store_id, repo->version, newdir);
02662         
02663         g_free(id);
02664         id = g_strndup(newdir->dir_id, 40);
02665         seaf_dir_free (newdir);
02666     }
02667 
02668 out:
02669     g_free (to_path_dup);
02670     seaf_dir_free(olddir);

02672 }

03704 // if no error and returned seafdir is NULL, then it means
03705 // file is not changed
03706 static SeafDir*
03707 get_seafdir_by_path (const char *repo_id,
03708                      int version,
03709                      const char *root_id,
03710                      const char *path,
03711                      GList *dir_ids,
03712                      GList **cur_dir_ids,
03713                      GError **error)
03714 {
03715     SeafDir *dir = NULL;
03716     SeafDirent *dent;
03717     const char *dir_id = root_id;
03718     char *name, *saveptr;
03719     char *tmp_path = NULL;
03720     GList *tmp = dir_ids;
03721 
03722     dir = seaf_fs_manager_get_seafdir (seaf->fs_mgr, repo_id, version, dir_id);
03723     if (!dir) {
03724         g_set_error (error, SEAFILE_DOMAIN, SEAF_ERR_DIR_MISSING, "directory is missing");
03725         goto out;
03726     }
03727 
03728     if (compare_or_add_id (tmp, cur_dir_ids, dir_id)) {
03729         seaf_dir_free (dir);
03730         dir = NULL;
03731         goto out;
03732     } else if (tmp) {
03733         tmp = tmp->next;
03734     }
03735 
03736     if (strcmp (path, ".") == 0 ||
03737         strcmp (path, "/") == 0) {
03738         goto out;
03739     } else {
03740         tmp_path = g_strdup (path);
03741     }
03742 
03743     name = strtok_r (tmp_path, "/", &saveptr);
03744     while (name != NULL) {
03745         GList *l;
03746         for (l = dir->entries; l != NULL; l = l->next) {
03747             dent = l->data;
03748

03750                 dir_id = dent->id;
03751                 break;
03752             }
03753         }
03754 
03755         if (!l) {
03756             g_set_error (error, SEAFILE_DOMAIN, SEAF_ERR_PATH_NO_EXIST,
03757                          "Path does not exists %s", path);
03758             seaf_dir_free (dir);
03759             dir = NULL;
03760             break;
03761         }
03762 
03763         if (compare_or_add_id (tmp, cur_dir_ids, dir_id)) {
03764             seaf_dir_free (dir);
03765             dir = NULL;
03766             goto out;
03767         } else if (tmp) {
03768             tmp = tmp->next;
03769         }
03770

03773         seaf_dir_free (prev);
03774 
03775         if (!dir) {
03776             g_set_error (error, SEAFILE_DOMAIN, SEAF_ERR_DIR_MISSING,
03777                          "directory is missing");
03778             break;
03779         }
03780 
03781         name = strtok_r (NULL, "/", &saveptr);
03782     }
03783 
03784 out:
03785     g_free (tmp_path);

04021 static gboolean
04022 detect_rename_revision (SeafRepo *repo,
04023                         SeafCommit *commit,
04024                         const char *path,
04025                         char **parent_id,
04026                         char **old_path)
04027 {
04028     GList *diff_res = NULL;
04029     SeafCommit *p1 = NULL;
04030     int rc;
04031     gboolean is_renamed = FALSE;
04032 
04033     while (*path == '/' && *path != 0)
04034         ++path;
04035 
04036     if (!commit->second_parent_id) {
04037         p1 = seaf_commit_manager_get_commit (seaf->commit_mgr,
04038                                              repo->id, repo->version,
04039                                              commit->parent_id);
04040         if (!p1) {
04041             seaf_warning ("Failed to get commit %.8s.\n", commit->parent_id);
04042             return FALSE;
04043         }
04044         /* Don't fold diff results for directories. We need to know a file was
04045          * renamed when its parent folder was renamed.
04046          */
04047         rc = diff_commits (p1, commit, &diff_res, FALSE);
04048         seaf_commit_unref (p1);
04049         if (rc < 0) {
04050             seaf_warning ("Failed to diff.\n");
04051             return FALSE;
04052         }
04053     } else {
04054         rc = diff_merge (commit, &diff_res, FALSE);
04055         if (rc < 0) {
04056             seaf_warning ("Failed to diff merge.\n");
04057             return FALSE;
04058         }
04059     }
04060 
04061     GList *ptr;
04062     DiffEntry *de;
04063     for (ptr = diff_res; ptr; ptr = ptr->next) {
04064         de = ptr->data;
04065         if (de->status == DIFF_STATUS_RENAMED && strcmp (de->new_name, path) == 0) {
04066             *old_path = g_strdup(de->name);
04067             is_renamed = TRUE;
04068             break;
04069         }
04070     }
04071     for (ptr = diff_res; ptr; ptr = ptr->next)
04072         diff_entry_free ((DiffEntry *)ptr->data);
04073     g_list_free (diff_res);
04074 
04075     if (!is_renamed)
04076         return FALSE;
04077 
04078     /* Determine parent commit containing the old path. */
04079     if (!commit->second_parent_id)
04080         *parent_id = g_strdup(commit->parent_id);
04081     else {
04082         if (path_exists_in_commit (repo, commit->parent_id, *old_path))
04083             *parent_id = g_strdup(commit->parent_id);
04084         else if (path_exists_in_commit (repo, commit->second_parent_id, *old_path))
04085             *parent_id = g_strdup(commit->second_parent_id);
04086         else {
04087             g_free (*old_path);
04088             *old_path = NULL;
04089             return FALSE;
04090         }
04091     }
04092 
04093     return TRUE;
04094 }

04291 collect_files_last_modified (SeafCommit *commit, void *vdata, gboolean *stop)
04292 {

04294     GError **error = data->error;
04295     SeafDirent *dent = NULL;
04296     char *file_id = NULL;
04297     SeafDir *dir = NULL;
04298     GList *ptr;
04299     gboolean ret = TRUE;
04300 
04301     data->current_commit = commit;
04302     dir = seaf_fs_manager_get_seafdir_by_path (seaf->fs_mgr,
04303                                                data->repo->store_id,

04310             *stop = TRUE;

04313         } else {
04314             g_clear_error (error);
04315         }
04316     }
04317

04320          * modified in the previous commit;
04321          */
04322         *stop = TRUE;

04324     }
04325

04327         dent = ptr->data;

04330             if (strcmp(file_id, dent->id) != 0) {
04331                 g_hash_table_remove (data->current_file_id_hash, dent->name);
04332             } else {
04333                 gint64 *ctime = g_new (gint64, 1);
04334                 *ctime = commit->ctime;
04335                 g_hash_table_replace (data->last_modified_hash, g_strdup(dent->name), ctime);
04336             }

04340             *stop = TRUE;
04341             goto out;
04342         }
04343     }
04344 
04345     /* Files not found in the current commit are last modified in the previous
04346      * commit */
04347     g_hash_table_foreach_remove (data->current_file_id_hash,
04348                                  check_non_existing_files, data);
04349 
04350     if (g_hash_table_size(data->current_file_id_hash) == 0) {
04351         /* All files under this diretory have been calculated  */
04352         *stop = TRUE;
04353         goto out;
04354     }
04355 
04356 out:
04357     seaf_dir_free (dir);
04358 
04359     return ret;
04360 }

04611 static int
04612 find_deleted_recursive (SeafRepo *repo,
04613                         SeafDir *d1,
04614                         SeafDir *d2,
04615                         const char *base,
04616                         SeafCommit *child,
04617                         SeafCommit *parent,
04618                         GHashTable *entries)
04619 {
04620     GList *p1, *p2;
04621     SeafDirent *dent1, *dent2;
04622     int res, ret = 0;
04623 
04624     p1 = d1->entries;
04625     p2 = d2->entries;
04626 
04627     /* Since dirents are sorted in descending order, we can use merge
04628      * algorithm to find out deleted entries.
04629      * Deleted entries are those:
04630      * 1. exists in d2 but absent in d1.
04631      * 2. exists in both d1 and d2 but with different type.
04632      */
04633 
04634     while (p1 && p2) {
04635         dent1 = p1->data;
04636         dent2 = p2->data;
04637 
04638         res = g_strcmp0 (dent1->name, dent2->name);
04639         if (res < 0) {
04640             /* exists in d2 but absent in d1. */
04641             add_deleted_entry (repo, entries, dent2, base, child, parent);
04642             p2 = p2->next;
04643         } else if (res == 0) {
04644             if ((dent1->mode & S_IFMT) != (dent2->mode & S_IFMT)) {
04645                 /* both exists but with diffent type. */
04646                 add_deleted_entry (repo, entries, dent2, base, child, parent);
04647             } else if (S_ISDIR(dent1->mode)) {
04648                 SeafDir *n1 = seaf_fs_manager_get_seafdir_sorted (seaf->fs_mgr,
04649                                                                   repo->id,
04650                                                                   repo->version,
04651                                                                   dent1->id);
04652                 if (!n1) {
04653                     seaf_warning ("Failed to find dir %s.\n", dent1->id);
04654                     return -1;
04655                 }
04656 
04657                 SeafDir *n2 = seaf_fs_manager_get_seafdir_sorted (seaf->fs_mgr,
04658                                                                   repo->id,
04659                                                                   repo->version,
04660                                                                   dent2->id);
04661                 if (!n2) {
04662                     seaf_warning ("Failed to find dir %s.\n", dent2->id);
04663                     seaf_dir_free (n1);
04664                     return -1;
04665                 }
04666 
04667                 char *new_base = g_strconcat (base, dent1->name, "/", NULL);
04668                 ret = find_deleted_recursive (repo, n1, n2, new_base,
04669                                               child, parent, entries);
04670                 g_free (new_base);
04671                 seaf_dir_free (n1);
04672                 seaf_dir_free (n2);
04673                 if (ret < 0)
04674                     return ret;
04675             }
04676             p1 = p1->next;
04677             p2 = p2->next;
04678         } else {
04679             p1 = p1->next;
04680         }
04681     }
04682 
04683     for ( ; p2 != NULL; p2 = p2->next) {
04684         dent2 = p2->data;
04685         add_deleted_entry (repo, entries, dent2, base, child, parent);
04686     }
04687
