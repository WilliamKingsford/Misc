02061 static size_t
02062 send_block_callback (void *ptr, size_t size, size_t nmemb, void *userp)
02063 {
02064     size_t realsize = size *nmemb;
02065     SendBlockData *data = userp;
02066     HttpTxTask *task = data->task;
02067     int n;
02068 
02069     if (task->state == HTTP_TASK_STATE_CANCELED)
02070         return CURL_READFUNC_ABORT;
02071 
02072     n = seaf_block_manager_read_block (seaf->block_mgr,
02073                                        data->block,
02074                                        ptr, realsize);
02075     if (n < 0) {
02076         seaf_warning ("Failed to read block %s in repo %.8s.\n",
02077                       data->block_id, task->repo_id);
02078         task->error = HTTP_TASK_ERR_BAD_LOCAL_DATA;
02079         return CURL_READFUNC_ABORT;
02080     }

02082     /* Update global transferred bytes. */
02083     g_atomic_int_add (&(seaf->sync_mgr->sent_bytes), n);
02084 
02085     /* If uploaded bytes exceeds the limit, wait until the counter
02086      * is reset. We check the counter every 100 milliseconds, so we
02087      * can waste up to 100 milliseconds without sending data after
02088      * the counter is reset.
02089      */
02090     while (1) {
02091         gint sent = g_atomic_int_get(&(seaf->sync_mgr->sent_bytes));
02092         if (seaf->sync_mgr->upload_limit > 0 &&
02093             sent > seaf->sync_mgr->upload_limit)
02094             /* 100 milliseconds */
02095             G_USLEEP (100000);
02096         else
02097             break;
02098     }
02099 
02100     return n;
02101 }
