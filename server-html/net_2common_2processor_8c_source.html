00144 /*
00145  * processor->detached is set in two places:
00146  * 1. ccnet_peer_remove_process(), which is called when one processor is done;
00147  * 2. ccnet_proc_factory_shutdown_processors(), which is called when peer shutdown.
00148  *
00149  * There are two shutdown/done situation:
00150  * 1. processor thread is not running;
00151  * 2. processor thread was running, now it's a delayed shutdown.
00152  *
00153  * When ccnet_processor_done() is called while worker thread is running,
00154  * the current state is saved in processor->detached, processor->was_success.
00155  * After the thread is done, ccnet_processor_done() will be called again
00156  * with the same state.
00157  */
00158 
00159 void
00160 ccnet_processor_done (CcnetProcessor *processor, gboolean success)
00161 {
00162     if (processor->thread_running) {
00163         processor->delay_shutdown = TRUE;
00164         processor->was_success = success;
00165         return;
00166     }
00167 
00168     if (processor->state == STATE_IN_SHUTDOWN) {
00169         return;
00170     }
00171 
00172     processor->state = STATE_IN_SHUTDOWN;
00173     if (processor->failure == PROC_NOTSET)
00174         processor->failure = PROC_DONE;
00175     if (!processor->peer->is_local)
00176         ccnet_debug ("Processsor %s(%d) done %d\n", GET_PNAME(processor),
00177                      PRINT_ID(processor->id), success);
00178 
00179     if (!processor->detached && success)
00180     {
00181         if (!IS_SLAVE (processor)) {
00182             ccnet_processor_send_update (processor, SC_PROC_DONE, SS_PROC_DONE,
00183                                          NULL, 0);
00184         }
00185     }
00186 
00187     g_signal_emit (processor, signals[DONE_SIG], 0, success);
00188 
00189     if (!processor->detached) {
00190         ccnet_peer_remove_processor (processor->peer, processor);
00191     }
00192 
00193     ccnet_processor_release_resource (processor);
00194 
00195     ccnet_proc_factory_recycle (processor->session->proc_factory, processor);
00196 }
