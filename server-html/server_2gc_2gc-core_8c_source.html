00020 /*
00021  * The number of bits in the bloom filter is 4 times the number of all blocks.
00022  * Let m be the bits in the bf, n be the number of blocks to be added to the bf
00023  * (the number of live blocks), and k = 3 (closed to optimal for m/n = 4),
00024  * the probability of false-positive is
00025  *
00026  *     p = (1 - e^(-kn/m))^k = 0.15
00027  *
00028  * Because m = 4 * total_blocks >= 4 * (live blocks) = 4n, we should have p <= 0.15.
00029  * Put it another way, we'll clean up at least 85% dead blocks in each gc operation.
00030  * See http://en.wikipedia.org/wiki/Bloom_filter.
00031  *
00032  * Supose we have 8TB space, and the avg block size is 1MB, we'll have 8M blocks, then
00033  * the size of bf is (8M * 4)/8 = 4MB.
00034  *
00035  * If total_blocks is a small number (e.g. < 100), we should try to clean all dead blocks.
00036  * So we set the minimal size of the bf to 1KB.
00037  */
00038 static Bloom *
00039 alloc_gc_index ()
00040 {
00041     size_t size;
00042 
00043     size = (size_t) MAX(total_blocks << 2, 1 << 13);
00044     size = MIN (size, MAX_BF_SIZE);
00045 
00046     seaf_message ("GC index size is %u Byte.\n", (int)size >> 3);
00047 
00048     return bloom_create (size, 3, 0);
00049 }
