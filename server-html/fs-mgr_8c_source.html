02420 static gboolean
02421 verify_seafdir_v0 (const char *dir_id, const uint8_t *data, int len,
02422                    gboolean verify_id)
02423 {
02424     guint32 meta_type;
02425     guint32 mode;
02426     char id[41];
02427     guint32 name_len;
02428     char name[SEAF_DIR_NAME_LEN];
02429     const uint8_t *ptr;
02430     int remain;
02431     int dirent_base_size;
02432     SHA_CTX ctx;
02433     uint8_t sha1[20];
02434     char check_id[41];
02435 
02436     if (len < sizeof(SeafdirOndisk)) {
02437         seaf_warning ("[fs mgr] Corrupt seafdir object %s.\n", dir_id);
02438         return FALSE;
02439     }
02440 
02441     ptr = data;
02442     remain = len;
02443 
02444     meta_type = get32bit (&ptr);
02445     remain -= 4;
02446     if (meta_type != SEAF_METADATA_TYPE_DIR) {
02447         seaf_warning ("Data does not contain a directory.\n");
02448         return FALSE;
02449     }
02450 
02451     if (verify_id)
02452         SHA1_Init (&ctx);
02453 
02454     dirent_base_size = 2 * sizeof(guint32) + 40;
02455     while (remain > dirent_base_size) {
02456         mode = get32bit (&ptr);
02457         memcpy (id, ptr, 40);
02458         id[40] = '\0';
02459         ptr += 40;
02460         name_len = get32bit (&ptr);
02461         remain -= dirent_base_size;
02462         if (remain >= name_len) {
02463             name_len = MIN (name_len, SEAF_DIR_NAME_LEN - 1);
02464             memcpy (name, ptr, name_len);
02465             ptr += name_len;
02466             remain -= name_len;
02467         } else {
02468             seaf_warning ("Bad data format for dir objcet %s.\n", dir_id);
02469             return FALSE;
02470         }
02471 
02472         if (verify_id) {
02473             /* Convert mode to little endian before compute. */
02474             if (G_BYTE_ORDER == G_BIG_ENDIAN)
02475                 mode = GUINT32_SWAP_LE_BE (mode);
02476 
02477             SHA1_Update (&ctx, id, 40);
02478             SHA1_Update (&ctx, name, name_len);
02479             SHA1_Update (&ctx, &mode, sizeof(mode));
02480         }
02481     }
02482 
02483     if (!verify_id)
02484         return TRUE;
02485 
02486     SHA1_Final (sha1, &ctx);
02487     rawdata_to_hex (sha1, check_id, 20);
02488 
02489     if (strcmp (check_id, dir_id) == 0)
02490         return TRUE;
02491     else
