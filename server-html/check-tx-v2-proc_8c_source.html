00096                         strlen(peer->session_key),

00098

00100     
00101     /* encrypt the token with session key, including the trailing null byte */
00102     if (seafile_encrypt (&enc_out, &len, token, strlen(token) + 1, crypt) < 0) {
00103         seaf_warning ("[check tx v2] failed to encrypt token\n");
00104         goto out;
00105     }
00106 
00107     output = g_malloc (len * 2 + 1);
00108     rawdata_to_hex ((unsigned char *)enc_out, output, len);
00109     output[len * 2] = '\0';

00111     
00112 out:
00113     g_free (crypt);
