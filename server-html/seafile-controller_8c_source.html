00069     GString *buf = g_string_new(argv[0]);

00071         g_string_append_printf (buf, " %s", *ptr);

00077 
00078     if (pid == 0) {
00079         /* child process */
00080         execvp (argv[0], argv);
00081         seaf_warning ("failed to execvp %s\n", argv[0]);
00082         exit(-1);

00086             seaf_warning ("error when fork %s: %s\n", argv[0], strerror(errno));

00088             seaf_message ("spawned %s, pid %d\n", argv[0], pid);
00089
