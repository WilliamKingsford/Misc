00701              *      a       (with a treated as 32 bits)
00702              */
00703             if (pp >= parts + 3)
00704                 return (0);
00705             *pp++ = val;
00706             c = *++src;
00707         } else
00708             break;
00709     }
00710     /*
00711      * Check for trailing characters.
00712      */
00713     if (c != '\0' && !isspace(c))
00714         return (0);
00715     /*
00716      * Concoct the address according to
00717      * the number of parts specified.
00718      */
00719     n = pp - parts + 1;
00720     /* inet_pton() takes dotted-quad only.  it does not take shorthand. */
00721     if (pton && n != 4)
00722         return (0);
00723     switch (n) {
00724 
00725     case 0:
00726         return (0);             /* initial nondigit */
00727 
00728     case 1:                         /* a -- 32 bits */
00729         break;
00730 
00731     case 2:                         /* a.b -- 8.24 bits */
00732         if (parts[0] > 0xff || val > 0xffffff)
00733             return (0);
00734         val |= parts[0] << 24;
00735         break;
00736 
00737     case 3:                         /* a.b.c -- 8.8.16 bits */
00738         if ((parts[0] | parts[1]) > 0xff || val > 0xffff)
00739             return (0);
00740         val |= (parts[0] << 24) | (parts[1] << 16);
00741         break;
00742 
00743     case 4:                         /* a.b.c.d -- 8.8.8.8 bits */
00744         if ((parts[0] | parts[1] | parts[2] | val) > 0xff)
00745             return (0);
00746         val |= (parts[0] << 24) | (parts[1] << 16) | (parts[2] << 8);
00747         break;
00748     }
00749     if (dst) {
00750         val = htonl(val);
00751         memcpy(dst, &val, INADDRSZ);
00752     }
00753     return (1);
00754 }

00756 /* int
00757  * inet_pton6(src, dst)
00758  *      convert presentation level address to network order binary form.
00759  * return:
00760  *      1 if `src' is a valid [RFC1884 2.2] address, else 0.
00761  * notice:
00762  *      (1) does not touch `dst' unless it's returning 1.
00763  *      (2) :: in a full address is silently ignored.
00764  * credit:
00765  *      inspired by Mark Andrews.
00766  * author:
00767  *      Paul Vixie, 1996.
00768  */
00769 static int
00770 inet_pton6 (const char *src, u_char *dst)
00771 {
00772     static const char xdigits_l[] = "0123456789abcdef",
00773         xdigits_u[] = "0123456789ABCDEF";
00774     u_char tmp[IN6ADDRSZ], *tp, *endp, *colonp;
00775     const char *xdigits, *curtok;
00776     int ch, saw_xdigit;
00777     u_int val;
00778 
00779     memset((tp = tmp), '\0', IN6ADDRSZ);
00780     endp = tp + IN6ADDRSZ;
00781     colonp = NULL;
00782     /* Leading :: requires some special handling. */
00783     if (*src == ':')
00784         if (*++src != ':')
00785             return (0);
00786     curtok = src;
00787     saw_xdigit = 0;
00788     val = 0;
00789     while ((ch = *src++) != '\0') {
00790         const char *pch;
00791 
00792         if ((pch = strchr((xdigits = xdigits_l), ch)) == NULL)
00793             pch = strchr((xdigits = xdigits_u), ch);
00794         if (pch != NULL) {
00795             val <<= 4;
00796             val |= (pch - xdigits);
00797             if (val > 0xffff)
00798                 return (0);
00799             saw_xdigit = 1;
00800             continue;
00801         }
00802         if (ch == ':') {
00803             curtok = src;
00804             if (!saw_xdigit) {
00805                 if (colonp)
00806                     return (0);
00807                 colonp = tp;
00808                 continue;
00809             } else if (*src == '\0')
00810                 return (0);
00811             if (tp + INT16SZ > endp)
00812                 return (0);
00813             *tp++ = (u_char) (val >> 8) & 0xff;
00814             *tp++ = (u_char) val & 0xff;
00815             saw_xdigit = 0;
00816             val = 0;
00817             continue;
00818         }
00819         if (ch == '.' && ((tp + INADDRSZ) <= endp) &&
00820             inet_pton4(curtok, tp, 1) > 0) {
00821             tp += INADDRSZ;
00822             saw_xdigit = 0;
00823             break;  /* '\0' was seen by inet_pton4(). */
00824         }
00825         return (0);
00826     }
00827     if (saw_xdigit) {
00828         if (tp + INT16SZ > endp)
00829             return (0);
00830         *tp++ = (u_char) (val >> 8) & 0xff;
00831         *tp++ = (u_char) val & 0xff;
00832     }
00833     if (colonp != NULL) {
00834         /*
00835          * Since some memmove()'s erroneously fail to handle
00836          * overlapping regions, we'll do the shift by hand.
00837          */
00838         const int n = tp - colonp;
00839         int i;
00840 
00841         if (tp == endp)
00842             return (0);
00843         for (i = 1; i <= n; i++) {
00844             endp[- i] = colonp[n - i];
00845             colonp[n - i] = 0;
00846         }
00847         tp = endp;
00848     }
00849     if (tp != endp)
00850         return (0);
00851     memcpy(dst, tmp, IN6ADDRSZ);
00852     return (1);
00853 }
